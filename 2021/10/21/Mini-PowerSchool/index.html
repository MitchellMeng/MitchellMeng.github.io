def signup():
  sign_up_window = tk.Toplevel(window)
  sign_up_window.geometry(‘300x200’)
  sign_up_window.title(‘Sign up’)
  tk.Label(sign_up_window, text=’User name:’).place(x=20, y=40)
  tk.Label(sign_up_window, text=’Password:’).place(x=25, y=80)
  tk.Label(sign_up_window, text=’Confirm password:’).place(x=15, y=120)
  var_user_name = tk.StringVar()
  entry_user_name = tk.Entry(sign_up_window, textvariable=var_user_name)
  entry_user_name.place(x=100, y=40)

  var_user_pass1 = tk.StringVar()
  entry_password1 = tk.Entry(sign_up_window, textvariable=var_user_pass1)
  entry_password1.place(x=100, y=80)

  var_user_pass2 = tk.StringVar()
  entry_password2 = tk.Entry(sign_up_window, textvariable=var_user_pass2)
  entry_password2.place(x=130, y=120)

  def confirm():
    user_name = var_user_name.get()
    user_pass1 = var_user_pass1.get()
    user_pass2 = var_user_pass2.get()
    cpath = user_name + “.xlsx”
    cpassword = user_pass1
    mpath1 = r”D:\桌面\仿powerschool\“ + cpath
    mpath2 = r”D:\桌面\仿powerschool\数据库”
    if user_pass1 == user_pass2:
      tk.messagebox.showinfo(message=’You have successfully signed up’)
      sign_up_window.destroy()
      userfile = xlsxwriter.Workbook(cpath)
      worksheet = userfile.add_worksheet(u”密码”)
      worksheet.write(“A1”,cpassword)
      userfile.close()
      shutil.move(mpath1,mpath2)
    else:
      tk.messagebox.showinfo(message=’The two passwords are different, please check’)
  def cancel():
    sign_up_window.destroy()
  
  tk.Button(sign_up_window, text=’Cancel’, command=cancel).place(x=50, y=160)
  tk.Button(sign_up_window, text=’Confirm’, command=confirm).place(x=130, y=160)
  sign_up_window.mainloop()

tk.Button(window,text=’Login’, command=login).place(x=75, y=410)
