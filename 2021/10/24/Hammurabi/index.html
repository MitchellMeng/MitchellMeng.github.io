Hammurabi

2021-10-24

Description

The Hammurabi is a very classical strategy game. You will be the ruler of ancient Samaria and plan the agriculture, cope with the diseases, deal with the rat and so on. At the end of the ten-year rule, you will get a feedback based on how well you develop the Samaria.

Code

import random
import math

def print_intro():
  print (‘’’Congrats, you are the newest ruler of ancient Samaria, elected for a ten year term
of office. Your duties are to distribute food, direct farming, and buy and sell land
as needed to support your people. Watch out for rat infestations and the resultant
plague! Grain is the general currency, measured in bushels. The following will help
you in your decisions:
 (a) Each person needs at least 20 bushels of grain per year to survive.
 (b) Each person can farm at most 10 acres of land.
 (c) It takes 2 bushels of grain to farm an acre of land.
 (d) The market price for land fluctuates yearly.
Rule wisely and you will be showered with appreciation at the end of your term. Rule
poorly and you will be kicked out of office!’’’)

def ask_to_buy_land(bushels, cost):
  ’’’Ask user how many bushels to spend buying land’’’
  acres = int(input(“How many acres will you buy? “))
  while acres * cost > bushels:
    print(f”O Great Hammurabi, we have but {bushels} bushels of grain!”)
    acres = int(input(“How many acres will you buy? “))
  return acres

def ask_to_sell_land(acres_owned):
  ’’’Ask user how much land they want to sell’’’
  acres = int(input(“How many acres of land will you sell? “))
  while acres > acres_owned:
    print(f”O Great Hammurabi, we have but {acres_owned} acres of land!”)
    acres = int(input(“How many acres of land will you sell? “))
  return acres

def ask_to_feed(bushels_in_storage):
  ’’’Ask user how many bushels they want to use for feeding’’’
  bushels = int(input(“How many bushels will you use for feeding your citizens? “))
  while bushels > bushels_in_storage:
    print(f”O Great Hammurabi, we have but {bushels_in_storage} bushels of grain!”)
    bushels = int(input(“How many bushels will you use for feeding your citizens? “))
  return bushels

def ask_to_cultivate(acres_owned, population, bushels_in_storage):
  ’’’Ask user how much land they want to plant seed in’’’
  acres = int(input(“How many acres of land will you plant seed in? “))
  while True:
    if acres > acres_owned:
      print(f”O Great Hammurabi, we have but {acres_owned} acre of land!”)
      acres = int(input(“How many acres of land will you plant seed in? “))
      continue
    if acres * 2 > bushels_in_storage:
      print(f”O Great Hammurabi, we have but {bushels_in_storage} bushels of grain!”)
      acres = int(input(“How many acres of land will you plant seed in? “))
      continue
    if acres > population * 10:
      print(f”O Great Hammurabi, we have but {population} people to cultivate the land!”)
      acres = int(input(“How many acres of land will you plant seed in? “))
      continue
    break
  return acres

def is_plague():
  ’’’If there is a plague’’’
  return random.random() <= 0.15

def num_starving(population, bushels_to_feed):
  ’’’How many people starved’’’
  if population * 20 > bushels_to_feed:
    people_starved = math.floor(population - bushels_to_feed / 20)
    if people_starved > math.floor(population * 0.45):
      print(“O Hammurabi, there were too many people died because of starving.”
        ” You are thrown out of office!”)
      end_game = True
    end_game = False
  else:
    people_starved = 0
    end_game = False
  return (people_starved, end_game)

def num_immigrants(land, grain_in_storage, population, num_starving):
  ’’’How many people came to the city’’’
  if num_starving > 0:
    immigrants = 0
  else:
    immigrants = math.floor((20 * land + grain_in_storage) / (100 * population + 1))
  return immigrants

def get_harvest(acre_planted, harvest_per_acre):
  ’’’How good the harvest is’’’
  bushels_harvested = harvest_per_acre * acre_planted
  return bushels_harvested

def do_rats_infest():
  ’’’Whether or not rats ruin things.’’’
  return random.random() < 0.4

def percent_destroyed():
  ’’’Extent of rat damage.’’’
  return random.randint(1, 3) / 10

def price_of_land():
  ’’’How much land costs next year.’’’
  return random.randint(16, 22)

def result(population, starved, acre, bushels):
  print(“O Great Hammurabi! You have successfully finished your ten year term of office.”)
  print(f”At the end of the tenth year, there are {population} people in the city.”)
  print(f”There are {starved} people starved in total.”)
  print(f”There are {acre} acres of land.”)
  print(f”There are {bushels} bushels of grain left.”)

def end():
  print(“———————————————————————————“)
  print(“Thanks for playing Hammurabi!”)

def Hammurabi():
  starved = 0
  immigrants = 5
  population = 100
  harvest = 3000
  bushels_per_acre = 3
  rats_ate = 200
  bushels_in_storage = 2800
  acres_owned = 1000
  cost_per_acre = 19
  plague_deaths = 0

  total_starved = 0
  acres_start = 1000

  print_intro()

  for i in range(1,11):
    print(“”)
    print(“O Great Hammurabi!”)
    print(f”You are in year {i} of your ten year rule.”)
    print(f”In the previous year {starved} people starved to death.”)
    print(f”In the previous year {immigrants} people entered the kingdom.”)
    print(f”The population is now {population}.”)
    print(f”We harvested {harvest} bushels as {bushels_per_acre} bushels per acre.”)
    print(f”Rats destroyed {rats_ate} bushels, leaving {bushels_in_storage} bushels in storage.”)
    print(f”The city owns {acres_owned} acres of land.”)
    print(f”Land is currently worth {cost_per_acre} bushels per acre.”)
    print(f”There were {plague_deaths} deaths from the plague.”)
    land_buy = ask_to_buy_land(bushels_in_storage, cost_per_acre)
    if land_buy == 0:
      land_sell = ask_to_sell_land(acres_owned)
      acres_owned -= land_sell
      bushels_in_storage += land_sell * cost_per_acre
    else:
      acres_owned += land_buy
      bushels_in_storage -= land_buy * cost_per_acre
    
    bushels_to_feed = ask_to_feed(bushels_in_storage)
    bushels_in_storage -= bushels_to_feed
    acres_planted = ask_to_cultivate(acres_owned, population, bushels_in_storage)
    bushels_in_storage -= 2 * acres_planted

    if is_plague():
      plague_deaths = population // 2
      population -= plague_deaths
    else:
      plague_deaths = 0
    
    starved, endgame = num_starving(population, bushels_to_feed)
    total_starved += starved
    population -= starved
    
    immigrants = num_immigrants(acres_owned, bushels_in_storage, population, starved)
    population += immigrants
    
    bushels_per_acre = random.randint(1, 8)
    harvest = get_harvest(acres_planted, bushels_per_acre)
    bushels_in_storage += harvest

    if do_rats_infest():
      rats_ate = int(bushels_in_storage * percent_destroyed())
      bushels_in_storage -= rats_ate
    else:
      rats_ate = 0

    cost_per_acre = price_of_land()

    if endgame or population == 0:
      endgame = True
      year = i
      break

  if endgame:
    print(f”——————–GAME OVER——————–”)
    print(f”   In year {year}, you lose your authority   “)
    print(f”     Thanks for playing Hammurabi!     “)
  else:
    result(population, total_starved, acres_owned, bushels_in_storage)
    if total_starved == 0 and acres_owned >= acres_start:
      print(“You did a good job! Ancient Samaria is more prosperous under your administration.”)
      end()
    elif total_starved <= 30 or acres_owned >= acres_start * 0.8:
      print(“The development of Ancient Samaria went down a little bit, try your best next time.”)
      end()
    elif total_starved <= 60 or acres_owned >= acres_start * 0.5:
      print(“Ancient Samaria is not prosperous than before. People doesn’t like you.”)
      end()
    else:
      print(“People really hate you due to the unreasonable administration! “)
      end()
