package frc.robot;

import edu.wpi.first.wpilibj.TimedRobot;
import edu.wpi.first.wpilibj.SpeedControllerGroup;
import edu.wpi.first.wpilibj.Joystick;
import edu.wpi.first.wpilibj.Solenoid;
import edu.wpi.first.wpilibj.drive.DifferentialDrive;
import com.ctre.phoenix.motorcontrol.ControlMode;
import com.ctre.phoenix.motorcontrol.can.TalonFX;
import com.ctre.phoenix.motorcontrol.can.VictorSPX;
import com.revrobotics.CANSparkMax;
import com.revrobotics.CANSparkMaxLowLevel.MotorType;
import edu.wpi.first.networktables.NetworkTable;
import edu.wpi.first.networktables.NetworkTableEntry;
import edu.wpi.first.networktables.NetworkTableInstance;
import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
import edu.wpi.first.wpilibj.AnalogOutput;
import edu.wpi.first.wpilibj.I2C;
import edu.wpi.first.wpilibj.Jaguar;
import edu.wpi.first.wpilibj.PWM;
import edu.wpi.first.wpilibj.util.Color;
import com.revrobotics.ColorSensorV3;
import com.revrobotics.ColorMatchResult;
import com.revrobotics.ColorMatch;
import edu.wpi.first.wpilibj.Timer;
import edu.wpi.first.wpilibj.CameraServer;

public class Robot extends TimedRobot {
 private DifferentialDrive m_myRobot;
 private static final int leftDeviceID = 3;
 private static final int rightDeviceID = 2;
 private static final int leftDeviceID2 = 4;
 private static final int rightDeviceID2 = 1;
 
 private CANSparkMax m_leftMotor;
 private CANSparkMax m_rightMotor;
 private CANSparkMax m_leftMotor2;
 private CANSparkMax m_rightMotor2;
 Solenoid S1 = new Solenoid(7);
 Solenoid S2 = new Solenoid(6);
 private Joystick gamePad = new Joystick(0);
 private Joystick baseControl = new Joystick(1);
 VictorSPX intakerollerright = new VictorSPX(5);
 VictorSPX intakerollerleft = new VictorSPX(12);
 TalonFX falcon1 = new TalonFX(14);
 TalonFX falson2 = new TalonFX(15);
 VictorSPX mainturret = new VictorSPX(8);
 VictorSPX shooterturning = new VictorSPX(6);
 VictorSPX zdzsb1 = new VictorSPX(7);
 VictorSPX zdzsb2 = new VictorSPX(13);
 VictorSPX ballcatcher = new VictorSPX(10);
 NetworkTable table = NetworkTableInstance.getDefault().getTable(“limelight”);
 NetworkTableEntry tx = table.getEntry(“tx”);
 NetworkTableEntry ty = table.getEntry(“ty”);
 NetworkTableEntry ta = table.getEntry(“ta”);
 double kp = 0.02;
 double ki = 0.03;
 double kd = 0.04;
 I2C.Port i2cPort = I2C.Port.kOnboard;
 ColorSensorV3 m_colorSensor = new ColorSensorV3(i2cPort);
 Color detectedColor = m_colorSensor.getColor();
 Timer time = new Timer();
 double adjust = 0.02;
 boolean zdzsb3 = false;
 private final ColorMatch m_colorMatcher = new ColorMatch();
 private final Color kYellowTarget = ColorMatch.makeColor(0.361, 0.524, 0.113);

 @Override
 public void robotInit() {
  m_leftMotor = new CANSparkMax(leftDeviceID, MotorType.kBrushless);
  m_rightMotor = new CANSparkMax(rightDeviceID, MotorType.kBrushless);
  m_leftMotor2 = new CANSparkMax(leftDeviceID2, MotorType.kBrushless);
  m_rightMotor2 = new CANSparkMax(rightDeviceID2, MotorType.kBrushless);
  
  SpeedControllerGroup m_left = new SpeedControllerGroup(m_leftMotor, m_leftMotor2);
  SpeedControllerGroup m_right = new SpeedControllerGroup(m_rightMotor, m_rightMotor2);
  time.start();
  time.reset();
  m_myRobot = new DifferentialDrive(m_left, m_right);
  m_myRobot.setExpiration(Robot.kDefaultPeriod);
  m_colorMatcher.addColorMatch(kYellowTarget);
  CameraServer.getInstance().startAutomaticCapture();
 }

 @Override
 public void robotPeriodic() {
 }

 @Override
 public void autonomousInit() {

 }
 @Override

 public void autonomousPeriodic() {
  m_myRobot.setSafetyEnabled(false);
  m_myRobot.arcadeDrive(0.3, -0.085);
  time.delay(3);
  m_myRobot.arcadeDrive(0, 0);
  falcon1.set(ControlMode.PercentOutput, 1);
  falson2.set(ControlMode.PercentOutput, -1);
  time.delay(2);
  falcon1.set(ControlMode.PercentOutput, 1);
  falson2.set(ControlMode.PercentOutput, -1);
  S2.set(true);
  zdzsb1.set(ControlMode.PercentOutput, -0.8);
  zdzsb2.set(ControlMode.PercentOutput, 0.8);
  ballcatcher.set(ControlMode.PercentOutput, 0.8);
  mainturret.set(ControlMode.PercentOutput,0.15);
  time.delay(11);
 }

 @Override
 public void teleopPeriodic() {

  m_myRobot.arcadeDrive(baseControl.getY(),-baseControl.getZ() * 0.75);
  boolean ballcatcher = baseControl.getRawButton(5);
  boolean ballcatcherback = baseControl.getRawButton(3);
  boolean ballshooter = baseControl.getRawButton(1);
  boolean PID = gamePad.getRawButton(2);
  boolean intakeReverse = baseControl.getRawButton(4);
  boolean deliver = baseControl.getRawButton(2);
  boolean reverse = baseControl.getRawButton(6);
  boolean plateForm = gamePad.getRawButton(1);
  Color color = m_colorSensor.getColor();
  String colorString ;
  Color detectedColor = m_colorSensor.getColor();
  ColorMatchResult match = m_colorMatcher.matchClosestColor(detectedColor);
  
  if (intakeReverse == true){
   intakerollerleft.set(ControlMode.PercentOutput, -0.8);
   intakerollerright.set(ControlMode.PercentOutput, 0.8);
  }

  if (ballcatcher == true){
   S1.set(true);
   zdzsb3 = true;
  }else if (ballcatcherback == true){
   S1.set(false);
   zdzsb3 = false;
  }

  if (zdzsb3==true) {
   mainturret.set(ControlMode.PercentOutput,0.1);
   if (reverse == true){
    mainturret.set(ControlMode.PercentOutput, -0.2);
    time.delay(0.3);
    mainturret.set(ControlMode.PercentOutput, 0);
   }
  }
  else if (zdzsb3=false){
   mainturret.set(ControlMode.PercentOutput, 0);
  }

  if (PID == true){
   NetworkTableInstance.getDefault().getTable(“limelight”).getEntry(“ledMode”).setNumber(3);
   double x = tx.getDouble(0.0);
   double target = 0;
   double a = x - target;
   double xp = akp;
   double inte = 0;
   inte = inte + a;
   double xi = inteki;
   double error2 = 0;
   double diff = error2 - a;
   error2 = a;
   double xd = diff*kd;
   adjust = xp + xi + xd;
   if (adjust>0.15){
    adjust = 0.15;
   }
   else if (adjust<-0.15){
    adjust = -0.15;
   }
   shooterturning.set(ControlMode.PercentOutput, adjust);
  }

  else if (gamePad.getRawButton(1)==true){
   NetworkTableInstance.getDefault().getTable(“limelight”).getEntry(“ledMode”).setNumber(3);
   shooterturning.set(ControlMode.PercentOutput, gamePad.getZ()/8);
  }
  else{
   shooterturning.set(ControlMode.PercentOutput, 0);
  }

  if (ballshooter == true){
   falcon1.set(ControlMode.PercentOutput, 1);
   falson2.set(ControlMode.PercentOutput, -1);
   time.delay(1.5);
   S2.set(true);
   while(true){
    zdzsb1.set(ControlMode.PercentOutput, -0.8);
    zdzsb2.set(ControlMode.PercentOutput, 0.8);
    ballcatcher.set(ControlMode.PercentOutput, 0.8);
    reverse = baseControl.getRawButton(6);
    ballshooter = baseControl.getRawButton(1);
    if (PID == true){
     NetworkTableInstance.getDefault().getTable(“limelight”).getEntry(“ledMode”).setNumber(3);
     double x = tx.getDouble(0.0);
     double target = 0;
     double a = x - target;
     double xp = akp;
     double inte = 0;
     inte = inte + a;
     double xi = inteki;
     double error2 = 0;
     double diff = error2 - a;
     error2 = a;
     double xd = diff*kd;
     adjust = xp + xi + xd;
     if (adjust>0.15){
      adjust = 0.15;
     }
     else if (adjust<-0.15){
      adjust = -0.15;
     }
     shooterturning.set(ControlMode.PercentOutput, adjust);
    }
    else if (gamePad.getRawButton(1)==true){
     NetworkTableInstance.getDefault().getTable(“limelight”).getEntry(“ledMode”).setNumber(3);
     shooterturning.set(ControlMode.PercentOutput, gamePad.getZ()/8);
    }
    else{
     shooterturning.set(ControlMode.PercentOutput, 0);
    }

    if (reverse == true){
     mainturret.set(ControlMode.PercentOutput, -0.2);
     time.delay(0.3);
    }
    else{
     mainturret.set(ControlMode.PercentOutput,0.2);
    }
    if (ballshooter == false){
     break;
    }
   }
  }

  else{
   falcon1.set(ControlMode.PercentOutput, 0);
   falson2.set(ControlMode.PercentOutput, 0);
   zdzsb1.set(ControlMode.PercentOutput, 0);
   zdzsb2.set(ControlMode.PercentOutput, 0);
   ballcatcher.set(ControlMode.PercentOutput, 0);
  }
  
  double IR = m_colorSensor.getIR();
  if(m_colorSensor.getIR()>= 50 && deliver == true){
   NetworkTableInstance.getDefault().getTable(“limelight”).getEntry(“ledMode”).setNumber(1);
   intakerollerright.set(ControlMode.PercentOutput,-0.8);
   intakerollerleft.set(ControlMode.PercentOutput, 0.8);
   mainturret.set(ControlMode.PercentOutput,0.15);
   zdzsb1.set(ControlMode.PercentOutput, -0.6);
   zdzsb2.set(ControlMode.PercentOutput,0.6);
   ballcatcher.set(ControlMode.PercentOutput,0.8);
   S2.set(true);
   deliver = baseControl.getRawButton(4);
   IR = m_colorSensor.getIR();
  }
  else if (deliver == true){
   NetworkTableInstance.getDefault().getTable(“limelight”).getEntry(“ledMode”).setNumber(1);
   intakerollerright.set(ControlMode.PercentOutput,-0.8);
   intakerollerleft.set(ControlMode.PercentOutput, 0.8);
   mainturret.set(ControlMode.PercentOutput,0.15);
   ballcatcher.set(ControlMode.PercentOutput,0);
   zdzsb1.set(ControlMode.PercentOutput,0);
   zdzsb2.set(ControlMode.PercentOutput,0);
   S2.set(false);
  }
  else{
   intakerollerright.set(ControlMode.PercentOutput,0);
   intakerollerleft.set(ControlMode.PercentOutput, 0);
   S2.set(false);
   zdzsb1.set(ControlMode.PercentOutput, 0);
   zdzsb2.set(ControlMode.PercentOutput, 0);
   ballcatcher.set(ControlMode.PercentOutput,0);
  }

  double angleDegree = 34.3 + ty.getDouble(0.0);
  double limeHeight = 0.51;
  double targetHeight = 2;
  double angleRadian = Math.toRadians(angleDegree);
  double dist = (targetHeight-limeHeight)/Math.tan(angleRadian);
  SmartDashboard.putNumber(“distance”, dist);
  SmartDashboard.putNumber(“IR”, IR);
 }

 @Override
 public void testPeriodic() {
}
